import {graphql} from 'gatsby';
export const pageQuery = graphql`
  query NotFoundPageQuery {
    posts: allMdx(
      filter: {slug: {regex: "/blog/./i"}, frontmatter: {published: {eq: true}}}
      sort: {fields: fileAbsolutePath, order: DESC}
      limit: 3
    ) {
      nodes {
        frontmatter {
          title
        }
        fields {
          slug
        }
        id
        excerpt
      }
    }
  }
`;
